% Resume PDF typography tweaks (fonts set via pandoc -V variables).
\usepackage{ragged2e}
\AtBeginDocument{\RaggedRight}

\usepackage{enumitem}
\setlist{
  itemsep=2pt,
  topsep=4pt,
  parsep=0pt,
  partopsep=0pt
}

% Avoid awkward mid-word breaks in a resume.
\hyphenpenalty=10000
\exhyphenpenalty=10000

\usepackage{titlesec}
% Pandoc: # = section, ## = subsection, ### = subsubsection, ** = body bold.
% Size steps (10pt base): name ~25pt, sections 15pt, entries 12pt, inline bold 10pt.
\titleformat{\section}{\normalfont\huge\bfseries}{}{0em}{}
\titleformat{\subsection}{\normalfont\fontsize{15}{18}\selectfont\bfseries}{}{0em}{}
\titleformat{\subsubsection}{\normalfont\fontsize{12}{14}\selectfont\bfseries}{}{0em}{}
\titlespacing*{\section}{0pt}{0pt}{0.4em}
\titlespacing*{\subsection}{0pt}{1em plus .2em minus .2em}{0.35em plus .2em minus .2em}
\titlespacing*{\subsubsection}{0pt}{0.6em plus .2em minus .2em}{0.15em plus .2em minus .2em}
